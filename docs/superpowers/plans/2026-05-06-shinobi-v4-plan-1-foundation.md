```typescript
export const ROLES = [
  'shinobi_worker',
  'floor_manager',
  'sales_team_leader',
  'crm_manager',
] as const

export type Role = (typeof ROLES)[number]

export interface JwtPayload {
  sub: string    // user id (UUID)
  email: string
  name: string
  role: Role
  iat: number
  exp: number
}

// Higher rank = more access. sales_team_leader and crm_manager are peers.
const ROLE_RANK: Record<Role, number> = {
  shinobi_worker: 4,
  floor_manager: 3,
  sales_team_leader: 2,
  crm_manager: 2,
}

/** Returns true if userRole has at least the access level of required. */
export function hasRole(userRole: Role, required: Role): boolean {
  return ROLE_RANK[userRole] >= ROLE_RANK[required]
}
```

```typescript
import { SignJWT, jwtVerify } from 'jose'
import type { JwtPayload, Role } from '@shinobi/shared'
import { env } from './env.js'

const secret = new TextEncoder().encode(env.JWT_SECRET)
const ALG = 'HS256'

export async function signToken(payload: {
  sub: string
  email: string
  name: string
  role: Role
}): Promise<string> {
  return new SignJWT(payload as Record<string, unknown>)
    .setProtectedHeader({ alg: ALG })
    .setIssuedAt()
    .setExpirationTime('7d')
    .sign(secret)
}

export async function verifyToken(token: string): Promise<JwtPayload> {
  const { payload } = await jwtVerify(token, secret)
  return payload as unknown as JwtPayload
}
```

```typescript
import type { JwtPayload } from '@shinobi/shared'
import { verifyToken } from '../lib/jwt.js'

export interface TrpcContext {
  user: JwtPayload | null
}

export async function createContext(authHeader: string | undefined): Promise<TrpcContext> {
  if (!authHeader?.startsWith('Bearer ')) return { user: null }
  try {
    return { user: await verifyToken(authHeader.slice(7)) }
  } catch {
    return { user: null }
  }
}
```

- [ ] **Step 3: Create `apps/api/src/trpc/routers/auth.ts`**

```typescript
import { z } from 'zod'
import { eq } from 'drizzle-orm'
import { TRPCError } from '@trpc/server'
import bcrypt from 'bcryptjs'
import { router, publicProcedure, protectedProcedure } from '../router.js'
import { db } from '../../lib/db.js'
import { users } from '../../db/schema.js'
import { signToken } from '../../lib/jwt.js'

export const authRouter = router({
  login: publicProcedure
    .input(z.object({
      email:    z.string().email(),
      password: z.string().min(1),
    }))
    .mutation(async ({ input }) => {
      const [user] = await db
        .select()
        .from(users)
        .where(eq(users.email, input.email))

      const valid = user && await bcrypt.compare(input.password, user.passwordHash)
      if (!valid) throw new TRPCError({ code: 'UNAUTHORIZED', message: 'Invalid credentials' })

      const token = await signToken({
        sub:   user.id,
        email: user.email,
        name:  user.name,
        role:  user.role,
      })

      return {
        token,
        user: { id: user.id, email: user.email, name: user.name, role: user.role },
      }
    }),

  me: protectedProcedure.query(({ ctx }) => ctx.user),
})
```

```typescript
import type { JwtPayload } from '@shinobi/shared'

const KEY = 'shinobi_token'

export const getToken  = (): string | null =>
  typeof window !== 'undefined' ? localStorage.getItem(KEY) : null

export const setToken  = (t: string) => localStorage.setItem(KEY, t)
export const clearToken = () => localStorage.removeItem(KEY)

export function parseToken(token: string): JwtPayload | null {
  try {
    const [, payload] = token.split('.')
    return JSON.parse(atob(payload)) as JwtPayload
  } catch {
    return null
  }
}

export const isExpired = (p: JwtPayload) => p.exp * 1000 < Date.now()
```

- [ ] **Step 3: Create `apps/web/hooks/useAuth.ts`**

```typescript
'use client'
import { create } from 'zustand'
import type { JwtPayload } from '@shinobi/shared'
import { getToken, setToken, clearToken, parseToken, isExpired } from '@/lib/auth'

interface AuthStore {
  user:    JwtPayload | null
  token:   string | null
  login:   (token: string) => void
  logout:  () => void
  hydrate: () => void
}

export const useAuth = create<AuthStore>((set) => ({
  user:  null,
  token: null,

  login: (token) => {
    const user = parseToken(token)
    if (!user) return
    setToken(token)
    set({ user, token })
  },

  logout: () => {
    clearToken()
    set({ user: null, token: null })
  },

  hydrate: () => {
    const token = getToken()
    if (!token) return
    const user = parseToken(token)
    if (!user || isExpired(user)) { clearToken(); return }
    set({ user, token })
  },
}))
```

```typescript
'use client'
import { useState, useEffect } from 'react'
import { QueryClient, QueryClientProvider } from '@tanstack/react-query'
import { httpBatchLink } from '@trpc/client'
import { trpc } from '@/lib/trpc'
import { useAuth } from '@/hooks/useAuth'
import { getToken } from '@/lib/auth'

function TrpcProvider({ children }: { children: React.ReactNode }) {
  const [queryClient] = useState(() => new QueryClient({
    defaultOptions: { queries: { staleTime: 30_000, retry: 1 } },
  }))

  const [trpcClient] = useState(() =>
    trpc.createClient({
      links: [
        httpBatchLink({
          url: `${process.env.NEXT_PUBLIC_API_URL}/trpc`,
          headers: () => {
            const token = getToken()
            return token ? { authorization: `Bearer ${token}` } : {}
          },
        }),
      ],
    })
  )

  return (
    <trpc.Provider client={trpcClient} queryClient={queryClient}>
      <QueryClientProvider client={queryClient}>
        {children}
      </QueryClientProvider>
    </trpc.Provider>
  )
}

function AuthHydrator({ children }: { children: React.ReactNode }) {
  const hydrate = useAuth((s) => s.hydrate)
  useEffect(() => { hydrate() }, [hydrate])
  return <>{children}</>
}

export function Providers({ children }: { children: React.ReactNode }) {
  return (
    <TrpcProvider>
      <AuthHydrator>{children}</AuthHydrator>
    </TrpcProvider>
  )
}
```
